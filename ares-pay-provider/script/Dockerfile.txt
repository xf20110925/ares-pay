#version: 1.0.1

FROM centos-6.8-base-custom:latest
MAINTAINER   wandapeng "dapeng.wan@pintuibao.cn"
###########发布相关端口##########
EXPOSE 22
#EXPOSE 8080
###########安装##########
RUN mkdir -p /opt/ares-account/logs/
COPY libs/ /opt/ares-account/libs/
COPY config/ /opt/ares-account/config/
COPY bin/ /opt/ares-account/bin/
RUN chmod 755 /opt/ares-account/bin/*
COPY ares-pay-provider.jar /opt/ares-account/
WORKDIR /opt/ares-account/
###########创建数据卷目录##########
VOLUME    /opt/ /var/log/
###########配置服务自启动##########
COPY run.sh  /deploy/scripts/
RUN chmod 755 /deploy/scripts/*.sh
ENTRYPOINT ["/deploy/scripts/run.sh"]
